#### 获取url 指定参数

``` js
function getUrlParam(name) {
    var reg = new RegExp('(^|&)' + name + '=([^&]*)(&|$)', 'i');
    var param = window.location.search.substr(1).match(reg);
    if (param !== null) {
      return unescape(param[2]);
    } else {
      return '';
    }
  }
// getUrlParam('token')
```

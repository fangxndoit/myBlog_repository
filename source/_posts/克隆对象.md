#### 对象深拷贝

``` js
// 方式一
let obj = { a: 1, b: 2 }
let clone = { ...obj };
clone.b = 4;
// obj = { a: 1, b: 2}, clone = { a: 1, b: 4 }


// 方式二
let otherClone = Object.assign({}, obj);
otherClone.b = 6;
clone.b = 4;
// obj = { a: 1, b: 2}, otherClone = { a: 1, b: 6 }


//方式三
let obj = { a: 1, b: { c: 2 } };
let clone = JSON.parseJSON.stringify(obj));
clone.b.c = 4;
// obj = { a: 1, b: { c: 2 }}, clone = { a: 1, b: { c: 4 } }
```
